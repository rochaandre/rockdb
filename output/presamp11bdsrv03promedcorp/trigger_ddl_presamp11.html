CREATE OR REPLACE  trigger TRI_SEGAUDIT_DDL after DDL on schema
declare
  sql_text    ora_name_list_t;
  conteudoDDL clob;
  Texto       clob;
  n           number;

  cursor cr_historico (nome_objeto varchar2) is
    select CSQL_AUDIT
      from segaudit
     where cnameaudit = nome_objeto
     order by nnumeaudit desc;
begin
  ----------------------------------------------
  --         Gera Log de Comandos DDL         --
  ----------------------------------------------
  n := ora_sql_txt(sql_text);
  conteudoDDL := '';
  for i in 1 .. n loop
    conteudoDDL := conteudoDDL || sql_text(i);
  end loop;

  if (instr(upper(conteudoDDL),'COMPILE') = 0) AND (instr(upper(conteudoDDL),'SEGAUDIT') = 0) then
    open cr_historico(ora_dict_obj_name);
    fetch cr_historico into texto;
    if cr_historico%notfound or texto <> conteudoDDL then
      insert into SEGAUDIT
        (DDATAAUDIT, COS__AUDIT, CCUSEAUDIT, CHOSTAUDIT,
         CIP__AUDIT,
         CTERMAUDIT, COWNEAUDIT, CTYPEAUDIT, CNAMEAUDIT, CSYSEAUDIT, CSQL_AUDIT)
        values
        (sysdate, sys_context('USERENV', 'OS_USER'), sys_context('USERENV', 'CURRENT_USER'), sys_context('USERENV', 'HOST'),
         substr(SYS_CONTEXT ('USERENV', 'IP_ADDRESS'),1,50),
         sys_context('USERENV', 'TERMINAL'), ora_dict_obj_owner, ora_dict_obj_type, ora_dict_obj_name, ora_sysevent, conteudoDDL);
    end if;
    close cr_historico;
  end if;
end;

CREATE OR REPLACE  TRIGGER sys.cdc_alter_ctable_before
  BEFORE
    ALTER ON DATABASE
    BEGIN
      /* NOP UNLESS A TABLE OBJECT */
      IF dictionary_obj_type = 'TABLE'
      THEN
        sys.dbms_cdc_ipublish.change_table_trigger(dictionary_obj_owner,dictionary_obj_name,sysevent);
      END IF;
      END;
CREATE OR REPLACE  TRIGGER sys.cdc_create_ctable_after
  AFTER
    CREATE ON DATABASE
    BEGIN
      /* NOP UNLESS A TABLE OBJECT */
      IF dictionary_obj_type = 'TABLE'
      THEN
        sys.dbms_cdc_ipublish.change_table_trigger(dictionary_obj_owner,dictionary_obj_name,sysevent);
      END IF;
      END;
CREATE OR REPLACE  TRIGGER sys.cdc_create_ctable_before
  BEFORE
    CREATE ON DATABASE
    BEGIN
      /* NOP UNLESS A TABLE OBJECT */
      IF dictionary_obj_type = 'TABLE'
      THEN
        sys.dbms_cdc_ipublish.change_table_trigger(dictionary_obj_owner,dictionary_obj_name,'LOCK');
      END IF;
      END;
CREATE OR REPLACE  TRIGGER sys.cdc_drop_ctable_before
  BEFORE
    DROP ON DATABASE
    BEGIN
      /* NOP UNLESS A TABLE OBJECT */
      IF dictionary_obj_type = 'TABLE'
      THEN
        sys.dbms_cdc_ipublish.change_table_trigger(dictionary_obj_owner,dictionary_obj_name,sysevent);
      END IF;
      END;

CREATE OR REPLACE  trigger sys.xdb_pi_trig
BEFORE DROP OR TRUNCATE on DATABASE
BEGIN
  BEGIN
    IF (sys.is_vpd_enabled(sys.dictionary_obj_owner, sys.dictionary_obj_name, xdb.DBMS_XDBZ.IS_ENABLED_CONTENTS)) THEN
      xdb.XDB_PITRIG_PKG.pitrig_truncate(sys.dictionary_obj_owner, sys.dictionary_obj_name);
    ELSIF (sys.is_vpd_enabled(sys.dictionary_obj_owner, sys.dictionary_obj_name, xdb.DBMS_XDBZ.IS_ENABLED_RESMETADATA)) THEN
      xdb.XDB_PITRIG_PKG.pitrig_dropmetadata(sys.dictionary_obj_owner, sys.dictionary_obj_name);
    END IF;
  EXCEPTION
    WHEN OTHERS THEN
     null;
  END;
END;
